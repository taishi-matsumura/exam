 */

{
	// 現在時刻から乱数の種を生成
	srand((unsigned int)time(NULL));

	// 0～9の乱数を生成
	int omikuji = rand() ① ②;

	// おみくじの結果文字列を格納する変数
	char* result;

	③(omikuji ④ 0)
	{
		result = "大凶";
	}
	⑤(omikuji ⑥ 2)
	{
		result = "凶";
	}
	⑤(omikuji ⑥ 5)
	{
		result = "小吉";
	}
	⑤(omikuji ⑥ 8)
	{
		result = "中吉";
	}
	⑦
	{
		result = "大吉";
